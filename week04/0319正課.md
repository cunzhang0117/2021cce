```C
struct POINT {
    float x, y;
};
int main()
{
    struct POINT a;
}

```

### 02｜struct POINT a={4.1, 3.2}; 並利用 a.x 及 a.y 印出來。
```C
#include <stdio.h>
struct POINT {
    float x, y;
};
int main()
{
    struct POINT a={4.1, 3.1};
    printf("%f %f\n", a.x, a.y);
}
```

### 03｜
```C
#include <stdio.h>
struct POINT {
    float x, y;
};
int main()
{
    struct POINT a={4.1, 3.1};
    printf("%f %f\n", a.x, a.y);

    a.x=1;
    a.y=2;
    printf("%f %f\n", a.x, a.y);
}
```
