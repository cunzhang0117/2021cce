### 01｜讀入整數反序列印
```C
#include <stdio.h>
int main()
{
	int a[11], ans=0;
	for(int i=0; i<=10; i++)
	{
		scanf("%d", &a[i]);
		if(a[i]==0)
		{
			ans=i;
			break;
		}
		
	}
	for(int i=ans-1; i>=0; i--)
	{
		printf("%d ", a[i]);
	}
	printf("\n");
}
```

### 02｜A的B次方函數
```C
#include <stdio.h>
int MYPOWER(int a, int b)
{
	if(b==0) return 1;
	else return a*MYPOWER(a, b-1);
}

int main(void)
{
	int a,b;
	scanf("%d%d",&a,&b);
	printf("[%d]",MYPOWER(a,b));
	return 0;
}
```

### 03∣漸增數列相加
```C
#include <stdio.h>
int main()
{
	int a, ans=0;
	scanf("%d", &a);
	
	for(int i=1; i<a; i++)
	{
		ans+=i*(i+1);
	}
	printf("%d\n", ans);
}
```
