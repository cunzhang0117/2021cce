### 01｜*p+2
```C
#include <stdio.h>
void printfAll()
{
    for(int i=0; i<=4; i++)
        printf("%d ", a[i]);
    printf("\n");
}
}
int main()
{
    int a[5]={0, 10, 20, 30 ,40};
    printfAll();

    int *p=&a[2];
    *p=222;
    printfAll()

    p=p+2;
    *p=666;
    printfAll();
}
```

### 03｜p亂碼
```C
#include <stdio.h>
int a[5]={0, 10, 20, 30 ,40};
void printfAll()
{
    for(int i=0; i<=4; i++)
        printf("%d ", a[i]);
    printf("\n");
}
int main()
{
    printfAll();

    int *p=&a[2];
    *p=222;
    printfAll();
    printf("p心裡的小紙條記的值是:%d\n", p);

    p=p+2;
    *p=444;
    printfAll();
    printf("p心裡的小紙條記的值是:%d\n", p);

    p--;
    *p=333;
    printfAll();
    printf("p心裡的小紙條記的值是:%d\n", p);
}
```
