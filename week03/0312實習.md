### 01｜計算陣列的平方值
```C
#include <stdio.h>
int main()
{
	int N, a[10];
	scanf("%d", &N);
	for(int i=0; i<=N-1; i++)
	{
		scanf("%d", &a[i]);
		printf("%d,", a[i]*a[i]);
	}
	printf("\n");
}
```

### 02｜大小寫轉換
```C
#include <stdio.h>
int main()
{
	char a[10];
	scanf("%s", &a);
	
	for(int i=0; i<=9; i++)
	{
		if(a[i]>='0' && a[i]<='9') printf("%c", a[i]);
		else if(a[i]>='A'&&a[i]<='Z') printf("%c", a[i]+32);
		else if(a[i]>='a'&&a[i]<='z') printf("%c", a[i]-32);
		else if(a[i]=='\0') break;
	}
	printf("\n");
}
```
