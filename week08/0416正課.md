### 01｜字串宣告與印出
```C
#include <stdio.h>
char a[100][10];
int main()
{
	int n;
	scanf("%d", &n);
	
	for(int i=0; i<n; i++)
	{
		scanf("%s", a[i]);
	}
	
	for(int i=0; i<n; i++)
	{
		printf("%s\n", a[i]);
	}
}
```

### 02｜快速排列
```C
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
char a[100][10];
int cmp(const void *p1, const void *p2)
{
	char *s1 = (char*) p1;
	char *s2 = (char*) p2;
	return strcmp(s1, s2);
}
int main()
{
	int n;
	scanf("%d", &n);
	
	for(int i=0; i<n; i++)
	{
		scanf("%s", a[i]);
	}
	
	qsort(a,n,10,cmp);
	
	for(int i=0; i<n; i++)
	{
		printf("%s\n", a[i]);
	}
}
```

### 04｜
```C
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
char a[100][10], other[100];
int cmp(const void *p1, const void *p2)
{
	return strcmp((char*)p1, (char*)p2);
}
int main()
{
	int n;
	scanf("%d", &n);
	for(int i=0; i<n; i++)
	{
		scanf("%s", a[i]);
		gets(other);
	}
	
	qsort(a, n, 10, cmp);
	int ans=1;
	
	for(int i=0; i<n; i++)
	{
		if(strcmp(a[i], a[i+1])==0) ans++;
		else{
			printf("%s %d\n", a[i], ans);
			ans=1;
		}
	}
}
```

```C
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
char tree[1000000][32];
int cmp(const void *p1, const void *p2)
{
	return strcmp((char*) p1, (char*) p2);
}
int main()
{
	int T;
	float N=0;
	scanf("%d\n\n", &T);
	
	for(int i=0;  ;i++)
	{
		char* in = gets( tree[i] );
		if(in==NULL || strcmp(tree[i], "")==0)
		{
			N=i;
			break;
		}			
	}
	
	qsort(tree, N, 32, cmp);
	float ans=1;
	
	for(int i=0; i<N; i++)
	{
		if(strcmp(tree[i], tree[i+1])==0) ans++;
		else
		{
			printf("%s %0.4f\n", tree[i], 100*(ans/N));
			ans=1;
		}
	}
}
```
