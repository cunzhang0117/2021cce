### 01｜字串宣告與印出
```C
#include <stdio.h>
char a[100][10];
int main()
{
	int n;
	scanf("%d", &n);
	
	for(int i=0; i<n; i++)
	{
		scanf("%s", a[i]);
	}
	
	for(int i=0; i<n; i++)
	{
		printf("%s\n", a[i]);
	}
}
```

### 02｜快速排列
```C
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
char a[100][10];
int cmp(const void *p1, const void *p2)
{
	char *s1 = (char*) p1;
	char *s2 = (char*) p2;
	return strcmp(s1, s2);
}
int main()
{
	int n;
	scanf("%d", &n);
	
	for(int i=0; i<n; i++)
	{
		scanf("%s", a[i]);
	}
	
	qsort(a,n,10,cmp);
	
	for(int i=0; i<n; i++)
	{
		printf("%s\n", a[i]);
	}
}
```
