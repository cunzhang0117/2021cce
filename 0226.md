### 01找零錢
```C
#include <stdio.h>
int main()
{
	int n, sum=0; 
	scanf("%d", &n);
	
	for(int i=1; i<=n ;i++)
	{
		if(n%i==0)
			sum++;
	}
	printf("%d\n", sum);
}
```

### 找倍數
```C
#include <stdio.h>
int main()
{
	int a[10], sum=0;
	
	for(int i=0; i<10; i++)
	{
		scanf("%d", &a[i]);
		if(a[i]%3==0)
			sum++;
	}
	
	printf("%d\n", sum);
}
```

### 整數轉換為等級
```C
#include <stdio.h>
int main()
{
	int n;
	scanf("%d", &n);
	if(n>=90) printf("A\n");
	else if(n<90 && n>=80) printf("B\n");
	else if(n<80 && n>=60) printf("C\n");
	else printf("F\n");
}
```

### 分式化簡
```C
#include <stdio.h>
int main()
{
	int c, m, num;
	scanf("%d%d", &c, &m);
	
	for(int i=1; i<=c; i++)
	{
		if(c%i==0 && m%i==0)
			num=i;
	}
	
	printf("%d %d\n", c/num, m/num);
}
```
